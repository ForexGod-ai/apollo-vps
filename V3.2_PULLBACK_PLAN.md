   - Break de High anterior
   - Confirmă cu momentum candle
   - Detectează CHoCH High și CHoCH Low

   ```python
   choch_high = max(candles_in_choch['high'])

   fibo_50 = choch_low + (choch_high - choch_low) * 0.5
   ```

3. **Zone Entry:**
   - **Zone Top**: `fibo_50 + (tolerance * 10 pips)`
   - **Zone Bottom**: `fibo_50 - (tolerance * 10 pips)`

4. **Validare Pullback:**
   ```python
   # Prețul trebuie să revină spre zona de breakout
   if trend == "bullish":

   ```

```python
def calculate_choch_fibonacci(choch_data, trend):
    """
    Calculează Fibonacci 50% retracement după CHoCH
    
    Args:
        choch_data (dict): {
            'high': float,
            'low': float,
            'break_price': float,
            'direction': str
        }
        trend (str): 'bullish' sau 'bearish'
    
    Returns:
        dict: {
            'fibo_50': float,
            'zone_top': float,
            'zone_bottom': float,
            'tolerance_pips': float
        }
    """
    choch_high = choch_data['high']
    choch_low = choch_data['low']
    range_pips = abs(choch_high - choch_low) * 10000  # Convert to pips
    
    # Calculează Fibo 50%
    if trend == "bullish":
        fibo_50 = choch_low + (choch_high - choch_low) * 0.5
    else:  # bearish
        fibo_50 = choch_high - (choch_high - choch_low) * 0.5
    
    # Zone cu toleranță ±10 pips
    tolerance = 0.0010  # 10 pips pentru EUR, USD pairs
    zone_top = fibo_50 + tolerance
    zone_bottom = fibo_50 - tolerance
    
    return {
        'fibo_50': fibo_50,
        'zone_top': zone_top,
        'zone_bottom': zone_bottom,
        'tolerance_pips': 10,
        'choch_range_pips': range_pips
    }
```

#### **B. Funcție Nouă: `validate_pullback_entry()`**
```python
def validate_pullback_entry(current_price, fibo_data, trend, choch_time, current_time, max_candles=24):
    """
    Validează dacă prețul a făcut pullback la Fibo 50% și e moment de entry
    
    Args:
        current_price (float): Prețul curent
        fibo_data (dict): Date Fibonacci calculate
        trend (str): 'bullish' sau 'bearish'
        choch_time (datetime): Timpul CHoCH-ului
        current_time (datetime): Timpul curent
        max_candles (int): Max candles după CHoCH pentru pullback
    
    Returns:
        tuple: (bool is_valid, str reason)
    """
    # Check timeout
    time_diff = (current_time - choch_time).total_seconds() / 3600  # Hours
    if time_diff > max_candles:
        return False, f"TIMEOUT: {time_diff:.1f}h since CHoCH (max {max_candles}h)"
    
    fibo_50 = fibo_data['fibo_50']
    zone_top = fibo_data['zone_top']
    zone_bottom = fibo_data['zone_bottom']
    
    # BULLISH: Prețul trebuie să revină LA SAU SUB Fibo 50%
    if trend == "bullish":
        if current_price <= zone_top and current_price >= zone_bottom:
            return True, f"✅ PULLBACK VALID @ {current_price:.5f} (Fibo 50%: {fibo_50:.5f})"
        elif current_price > zone_top:
            return False, f"⏳ Waiting pullback (current: {current_price:.5f}, target: {fibo_50:.5f})"
        else:
            return False, f"❌ Price below zone (current: {current_price:.5f}, zone_bottom: {zone_bottom:.5f})"
    
    # BEARISH: Prețul trebuie să revină LA SAU PESTE Fibo 50%
    else:
        if current_price >= zone_bottom and current_price <= zone_top:
            return True, f"✅ PULLBACK VALID @ {current_price:.5f} (Fibo 50%: {fibo_50:.5f})"
        elif current_price < zone_bottom:
            return False, f"⏳ Waiting pullback (current: {current_price:.5f}, target: {fibo_50:.5f})"
        else:
            return False, f"❌ Price above zone (current: {current_price:.5f}, zone_top: {zone_top:.5f})"
```
